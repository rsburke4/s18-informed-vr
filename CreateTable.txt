CREATE TABLE User(
	Name VARCHAR(20),
	PRIMARY KEY(Name)
);

CREATE TABLE Posts(
	ID VARCHAR(7),
	Subreddit_ID VARCHAR(8),
	TimeCode INTEGER,
	Author VARCHAR(20),
	Title VARCHAR(100),
	Body VARCHAR(15000),
	Link VARCHAR(50),
	PRIMARY KEY (ID),
	FOREIGN KEY (Author) REFERENCES User(Name)
);

CREATE TABLE Parent_Post(
	ID VARCHAR(6),
	PRIMARY KEY (ID),
	FOREIGN KEY (ID) REFERENCES Posts(ID)
);

CREATE TABLE Child_Post(
	ID VARCHAR(7),
	Parent_ID VARCHAR(6),
	PRIMARY KEY (ID),
	FOREIGN KEY (ID) REFERENCES Posts(ID),
	FOREIGN KEY (Parent_ID) REFERENCES Parent_Post(ID)
);
